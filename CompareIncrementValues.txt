    // Without any subroutines, take input, store as Max. Halt if Max <= 0.
    // Compare Max vs Sum. If Sum < Max, take input Incr to increment Sum.
    // Add Sum + Incr. Output each incremented Sum. Loop until Sum > Max.
    // Restart program, halting if Max <= 0.

Start,      Clear                // Start program, Set AC = 0
            Store      Sum       // Sum = AC

            Input                // Get Max value
            Store      Max       // Max = AC
            SkipCond   800       // If Max > 0,
            Halt                 // Else, halt if Max <= 0
            Output     Max       // Output Max

incrLoop,   Input                // Get increment value
            Store      Incr      // Incr = AC
            Add        Sum       // Increment Sum
            Store      Sum       // Sum += Incr
            Output     Sum       // Output incremented sums
            
            Load       Sum       // AC = Sum
            Sub        Max       // AC = Sum - Max
            SkipCond   000       // If AC < 0,
            Jump       Start     // Else, Sum > Max, restart
            Jump       incLoop   // Then, Sum < Max, reloop

// Declarations:
Max,        Dec        000
Incr,       Dec        000
Sum,        Dec        000
            End        Start
